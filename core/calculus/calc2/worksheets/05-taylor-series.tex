\wsheader{Calc 2}{Taylor \& Maclaurin Series}{A}
\begin{problems}
\item Write the Maclaurin series for $\sin x$ through degree $7$.
\item Find the Maclaurin series for $f(x) = \dfrac{1}{1 + x^2}$.
\item Find the Taylor series of $\ln x$ centered at $a = 1$ (first four nonzero terms).
\item Use a known series to evaluate $\displaystyle \lim_{x \to 0} \dfrac{1 - \cos x}{x^2}$.
\item Estimate $\sin(0.1)$ using two nonzero terms of its Maclaurin series; give an error bound.
\end{problems}

\wsheader{Calc 2}{Taylor \& Maclaurin Series}{B}
\begin{problems}
\item Write the Maclaurin series for $\cos x$ through degree $6$.
\item Find a Maclaurin series for $x e^{-x^2}$.
\item Find the Taylor series of $\sqrt{x}$ centered at $a = 4$ (first three nonzero terms).
\item Use a known series to evaluate $\displaystyle \lim_{x \to 0} \dfrac{e^x - 1 - x}{x^2}$.
\item Estimate $e^{0.2}$ using terms through $x^3$ of its Maclaurin series; give an error bound.
\end{problems}

\anshdr{Calc 2}{Taylor \& Maclaurin Series}

\begin{answers}{A}
\item $\sin x = x - \tfrac{x^3}{6} + \tfrac{x^5}{120} - \tfrac{x^7}{5040} + \cdots$
\item Substitute $-x^2$ into $\tfrac{1}{1-u} = \sum u^n$: $\boxed{\sum_{n=0}^\infty (-1)^n x^{2n}}$, $|x|<1$.
\item Let $u = x - 1$. $\ln(1 + u) = u - \tfrac{u^2}{2} + \tfrac{u^3}{3} - \tfrac{u^4}{4} + \cdots$, so $\ln x = (x-1) - \tfrac{(x-1)^2}{2} + \tfrac{(x-1)^3}{3} - \tfrac{(x-1)^4}{4} + \cdots$.
\item $\cos x = 1 - \tfrac{x^2}{2} + O(x^4)$, so $1 - \cos x = \tfrac{x^2}{2} + O(x^4)$, and $\tfrac{1-\cos x}{x^2} \to \boxed{1/2}$.
\item $\sin(0.1) \approx 0.1 - \tfrac{(0.1)^3}{6} \approx 0.099833$. Alternating series error $\le \tfrac{(0.1)^5}{120} \approx 8.3 \times 10^{-8}$.
\end{answers}

\begin{answers}{B}
\item $\cos x = 1 - \tfrac{x^2}{2} + \tfrac{x^4}{24} - \tfrac{x^6}{720} + \cdots$
\item $e^{-x^2} = \sum_{n=0}^\infty \tfrac{(-x^2)^n}{n!}$, so $x e^{-x^2} = \displaystyle \sum_{n=0}^\infty \tfrac{(-1)^n x^{2n+1}}{n!}$.
\item $f(x) = x^{1/2}$ at $a = 4$: $f(4) = 2$, $f'(4) = 1/4$, $f''(4) = -1/32$. $\sqrt x = 2 + \tfrac{x-4}{4} - \tfrac{(x-4)^2}{64} + \cdots$.
\item $e^x = 1 + x + \tfrac{x^2}{2} + O(x^3)$, so $e^x - 1 - x = \tfrac{x^2}{2} + O(x^3)$, ratio $\to \boxed{1/2}$.
\item $e^{0.2} \approx 1 + 0.2 + \tfrac{0.04}{2} + \tfrac{0.008}{6} = 1.22133$. Lagrange bound $\le \tfrac{e^{0.2}(0.2)^4}{24} \le \tfrac{1.25 \cdot 0.0016}{24} \approx 8.3 \times 10^{-5}$.
\end{answers}
