\wsheader{Calc 3}{Gradient \& Directional Derivatives}{A}
\begin{problems}
\item Find $\grad f$ for $f(x,y) = x^2 y + y^3$.
\item Find the directional derivative of $f(x,y) = x^2 - y^2$ at $(1,2)$ in the direction of $\vc v = \langle 3, 4\rangle$.
\item Find the direction of maximum increase of $f(x,y,z) = xyz$ at $(1,2,3)$ and the magnitude of that maximum rate.
\item Find the tangent plane to the level surface $x^2 + y^2 - z^2 = 1$ at $(1, 1, 1)$.
\item For $f(x,y,z) = x e^{yz}$, compute $D_{\vc u} f(1, 0, 2)$ where $\vc u$ is the unit vector toward $(2, 1, 0)$.
\end{problems}

\wsheader{Calc 3}{Gradient \& Directional Derivatives}{B}
\begin{problems}
\item Find $\grad f$ for $f(x,y,z) = \sqrt{x^2 + y^2 + z^2}$.
\item Find the directional derivative of $f(x,y) = \ln(x^2 + y^2)$ at $(1, 2)$ toward $(0, 0)$.
\item At which point on the surface $x^2 + 2y^2 + 3z^2 = 6$ is the tangent plane parallel to the $xy$-plane?
\item Find the rate of change of $f(x,y) = xy + y^2$ at $(2, 3)$ in the direction making angle $\pi/3$ with positive $x$-axis.
\item Find the equation of the normal line to the surface $z = x^2 + y^2$ at the point $(1, 2, 5)$.
\end{problems}

\anshdr{Calc 3}{Gradient \& Directional Derivatives}

\begin{answers}{A}
\item $\grad f = \boxed{\langle 2xy, \, x^2 + 3y^2\rangle}$.
\item $\grad f = \langle 2x, -2y\rangle$, at $(1,2)$: $\langle 2, -4\rangle$. $\hat{\vc v} = \tfrac{1}{5}\langle 3, 4\rangle$. $D_{\vc u} f = \tfrac{1}{5}(6 - 16) = \boxed{-2}$.
\item $\grad f = \langle yz, xz, xy\rangle$, at $(1,2,3)$: $\langle 6, 3, 2\rangle$. Max increase direction: $\langle 6,3,2\rangle/\sqrt{49}$. Max rate $= \boxed{\sqrt{49} = 7}$.
\item $\grad F = \langle 2x, 2y, -2z\rangle$, at $(1,1,1)$: $\langle 2, 2, -2\rangle$. Plane: $\boxed{2(x-1) + 2(y-1) - 2(z-1) = 0}$ i.e.\ $x + y - z = 1$.
\item $\grad f = \langle e^{yz}, xz e^{yz}, xy e^{yz}\rangle$, at $(1,0,2)$: $\langle 1, 2, 0\rangle$. $\vc u = \tfrac{(2-1, 1-0, 0-2)}{\sqrt{1+1+4}} = \tfrac{\langle 1,1,-2\rangle}{\sqrt 6}$. $D_{\vc u} f = \tfrac{1}{\sqrt 6}(1 + 2 + 0) = \boxed{3/\sqrt 6 = \sqrt{6}/2}$.
\end{answers}

\begin{answers}{B}
\item Let $r = \sqrt{x^2+y^2+z^2}$. Then $\grad f = \boxed{\dfrac{\langle x, y, z\rangle}{\sqrt{x^2+y^2+z^2}}}$.
\item $\grad f = \tfrac{\langle 2x, 2y\rangle}{x^2+y^2}$, at $(1,2)$: $\tfrac{1}{5}\langle 2, 4\rangle$. Direction toward origin: $\tfrac{(0-1, 0-2)}{\sqrt 5} = \tfrac{1}{\sqrt 5}\langle -1, -2\rangle$. $D = \tfrac{1}{5\sqrt 5}(-2 - 8) = \boxed{-\tfrac{2}{\sqrt 5}}$.
\item Tangent plane parallel to $xy$-plane $\Leftrightarrow$ normal vertical $\Leftrightarrow$ $\grad F = \langle 2x, 4y, 6z\rangle$ has $x$-and-$y$ components zero $\Rightarrow x = 0, y = 0$. Then $3z^2 = 6 \Rightarrow z = \pm \sqrt 2$. Points: $\boxed{(0, 0, \pm\sqrt 2)}$.
\item $\grad f = \langle y, x + 2y\rangle$, at $(2,3)$: $\langle 3, 8\rangle$. $\vc u = \langle \cos(\pi/3), \sin(\pi/3)\rangle = \langle 1/2, \sqrt 3/2\rangle$. $D = 3/2 + 4\sqrt 3 = \boxed{3/2 + 4\sqrt 3}$.
\item $\grad F$ of $F = z - x^2 - y^2$ at $(1,2,5)$: $\langle -2, -4, 1\rangle$. Normal line: $\boxed{(x,y,z) = (1,2,5) + t\langle -2, -4, 1\rangle}$.
\end{answers}
