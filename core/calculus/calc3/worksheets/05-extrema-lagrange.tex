\wsheader{Calc 3}{Extrema \& Lagrange Multipliers}{A}
\begin{problems}
\item Find and classify the critical points of $f(x,y) = x^2 + y^2 - 4x - 6y + 5$.
\item Find and classify the critical points of $f(x,y) = xy(1 - x - y)$.
\item Find the absolute extrema of $f(x,y) = x^2 + y^2$ on $x^2 + y^2 \le 4$.
\item Use Lagrange multipliers to find the max/min of $f(x,y) = x + 2y$ on $x^2 + y^2 = 5$.
\item Find the dimensions of the rectangular box of maximum volume inscribed in the ellipsoid $x^2 + 4y^2 + 9z^2 = 9$ (Lagrange).
\end{problems}

\wsheader{Calc 3}{Extrema \& Lagrange Multipliers}{B}
\begin{problems}
\item Find and classify the critical points of $f(x,y) = x^3 + y^3 - 3xy$.
\item Find the local extrema of $f(x,y) = e^{x^2 - y^2}$.
\item Use Lagrange to extremize $f(x,y,z) = x + y + z$ subject to $x^2 + y^2 + z^2 = 3$.
\item Use Lagrange to find the minimum distance from the origin to the plane $x + 2y + 3z = 14$.
\item Maximize $f(x,y) = xy$ subject to $4x^2 + y^2 = 8$.
\end{problems}

\anshdr{Calc 3}{Extrema \& Lagrange Multipliers}

\begin{answers}{A}
\item $\grad f = \langle 2x - 4, 2y - 6\rangle = \vc 0$ at $(2, 3)$. $f_{xx} = 2$, $f_{yy} = 2$, $f_{xy} = 0$; $H = 4 > 0$, $f_{xx} > 0$, so \textbf{local min at $(2,3)$}, value $-8$.
\item $f_x = y(1 - 2x - y)$, $f_y = x(1 - x - 2y)$. Critical points: $(0,0), (1,0), (0,1), (1/3, 1/3)$. At $(1/3,1/3)$: $f_{xx} = -2/3$, $f_{yy} = -2/3$, $f_{xy} = 1/3$. $H = 4/9 - 1/9 > 0$, $f_{xx} < 0$, \textbf{local max}, value $1/27$. The other three are saddle points.
\item Interior critical point only at $(0,0)$, where $f = 0$. On boundary $x^2 + y^2 = 4$: $f = 4$. \textbf{Min $= 0$ at $(0,0)$; max $= 4$ on the circle.}
\item $\grad f = \langle 1, 2\rangle = \lambda \langle 2x, 2y\rangle$, so $x = 1/(2\lambda)$, $y = 1/\lambda$. Constraint: $\tfrac{1}{4\lambda^2} + \tfrac{1}{\lambda^2} = 5$, so $\lambda^2 = 1/4$, $\lambda = \pm 1/2$. Points $(1, 2)$ (\textbf{max} $f = 5$) and $(-1, -2)$ (\textbf{min} $f = -5$).
\item Maximize $V = 8xyz$ subject to $g = x^2 + 4y^2 + 9z^2 = 9$ ($x,y,z>0$). Lagrange equations: $8yz = 2\lambda x$, $8xz = 8\lambda y$, $8xy = 18\lambda z$. Multiplying these gives $x^2 = 3$, $4y^2 = 3$, $9z^2 = 3$, so $x = \sqrt 3$, $y = \sqrt 3/2$, $z = 1/\sqrt 3$. Maximum volume $V = 8 xyz = 8 \cdot \sqrt 3 \cdot \tfrac{\sqrt 3}{2} \cdot \tfrac{1}{\sqrt 3} = \boxed{4\sqrt 3}$.
\end{answers}

\begin{answers}{B}
\item $f_x = 3x^2 - 3y = 0 \Rightarrow y = x^2$; $f_y = 3y^2 - 3x = 0 \Rightarrow x = y^2 = x^4$, so $x(x^3 - 1) = 0 \Rightarrow x = 0$ or $1$. Points: $(0,0)$ and $(1,1)$. $f_{xx} = 6x, f_{yy} = 6y, f_{xy} = -3$. At $(0,0)$: $H = 0 - 9 < 0$, \textbf{saddle}. At $(1,1)$: $H = 36 - 9 > 0$, $f_{xx} > 0$, \textbf{local min}, value $-1$.
\item $f_x = 2x e^{x^2 - y^2}$, $f_y = -2y e^{x^2-y^2}$. Zero only at $(0,0)$. $f_{xx} = (2 + 4x^2) e^{x^2-y^2}$, $f_{yy} = (-2 + 4y^2)e^{x^2-y^2}$, $f_{xy} = -4xy e^{x^2-y^2}$. At $(0,0)$: $f_{xx} = 2$, $f_{yy} = -2$, $H = -4 < 0$, \textbf{saddle point}.
\item $\grad f = \lambda \grad g$: $1 = 2\lambda x$, etc., so $x = y = z = \tfrac{1}{2\lambda}$. Constraint $3 \cdot \tfrac{1}{4\lambda^2} = 3 \Rightarrow \lambda = \pm 1/2$. Points $(1,1,1)$ with $f = 3$ (\textbf{max}) and $(-1,-1,-1)$ with $f = -3$ (\textbf{min}).
\item Min $x^2 + y^2 + z^2$ s.t.\ $x + 2y + 3z = 14$. $\grad$: $2x = \lambda$, $2y = 2\lambda$, $2z = 3\lambda$ $\Rightarrow x = \lambda/2, y = \lambda, z = 3\lambda/2$. Constraint: $\lambda/2 + 2\lambda + 9\lambda/2 = 14 \Rightarrow 7\lambda = 14 \Rightarrow \lambda = 2$. Point $(1,2,3)$, distance $= \sqrt{1+4+9} = \boxed{\sqrt{14}}$.
\item $\grad f = \lambda \grad g$: $y = 8\lambda x$, $x = 2\lambda y$. Multiply: $xy = 16\lambda^2 xy \Rightarrow \lambda = \pm 1/4$. From $y = 2x$ (taking $\lambda = 1/4$, $x,y > 0$), constraint $4x^2 + 4x^2 = 8 \Rightarrow x = 1, y = 2$. Max $f = \boxed{2}$.
\end{answers}
