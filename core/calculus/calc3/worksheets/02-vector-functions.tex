\wsheader{Calc 3}{Vector Functions \& Space Curves}{A}
\begin{problems}
\item For $\rr(t) = \langle t, t^2, t^3 \rangle$, find $\rr'(t)$ and $\rr''(t)$.
\item Find the unit tangent $\vc T(t)$ for $\rr(t) = \langle \cos t, \sin t, t \rangle$ (helix).
\item Find the arc length of $\rr(t) = \langle \cos t, \sin t, t\rangle$ on $[0, 2\pi]$.
\item Compute the curvature of the circle $\rr(t) = \langle 3 \cos t, 3 \sin t, 0 \rangle$.
\item For $\rr(t) = \langle t, t^2, 0\rangle$, compute the curvature at $t = 0$ using $\kappa = \tfrac{\|\rr'\times\rr''\|}{\|\rr'\|^3}$.
\end{problems}

\wsheader{Calc 3}{Vector Functions \& Space Curves}{B}
\begin{problems}
\item For $\rr(t) = \langle e^t, e^{-t}, t\sqrt 2 \rangle$, find $\|\rr'(t)\|$.
\item Find the arc length of $\rr(t) = \langle e^t, e^{-t}, t\sqrt 2\rangle$ on $[0, 1]$.
\item For a particle with position $\rr(t) = \langle t, t^2, t^3\rangle$, find velocity, speed, and acceleration at $t=1$.
\item Compute the tangential component of acceleration for $\rr(t) = \langle t^2, t, 0\rangle$ at $t = 1$.
\item For the plane curve $y = \tfrac{1}{x}$, $x > 0$, find $\kappa(x)$ using $\kappa = \tfrac{|y''|}{(1+(y')^2)^{3/2}}$, and evaluate at $x = 1$.
\end{problems}

\vfill

\anshdr{Calc 3}{Vector Functions \& Space Curves}

\begin{answers}{A}
\item $\rr'(t) = \boxed{\langle 1, 2t, 3t^2 \rangle}$, $\rr''(t) = \boxed{\langle 0, 2, 6t \rangle}$.
\item $\rr'(t) = \langle -\sin t, \cos t, 1\rangle$, $\|\rr'(t)\| = \sqrt 2$. $\vc T = \boxed{\tfrac{1}{\sqrt 2}\langle -\sin t, \cos t, 1\rangle}$.
\item $L = \int_0^{2\pi} \sqrt 2 \dt = \boxed{2\sqrt 2 \pi}$.
\item Circle of radius $3$, so $\kappa = \boxed{1/3}$ (constant).
\item $\rr'(t) = \langle 1, 2t, 0\rangle$, $\rr''(t) = \langle 0, 2, 0\rangle$. $\rr'\times\rr'' = \langle 0, 0, 2\rangle$, $\|\rr'\times\rr''\| = 2$. $\|\rr'\|^3 = (1+4t^2)^{3/2}$. At $t = 0$: $\kappa = \boxed{2}$.
\end{answers}

\begin{answers}{B}
\item $\rr'(t) = \langle e^t, -e^{-t}, \sqrt 2\rangle$. $\|\rr'\|^2 = e^{2t} + e^{-2t} + 2 = (e^t + e^{-t})^2$, so $\|\rr'\| = \boxed{e^t + e^{-t}}$.
\item $L = \int_0^1 (e^t + e^{-t}) \dt = [e^t - e^{-t}]_0^1 = \boxed{e - 1/e}$.
\item $\vc v = \langle 1, 2t, 3t^2\rangle$, at $t=1$: $\boxed{\vc v(1) = \langle 1,2,3\rangle}$, speed $= \sqrt{14}$, $\vc a(1) = \boxed{\langle 0,2,6\rangle}$.
\item $\rr'(t) = \langle 2t, 1, 0\rangle$, $\rr''(t) = \langle 2, 0, 0\rangle$. $a_T = \dfrac{\rr'\dotp\rr''}{\|\rr'\|} = \dfrac{4t}{\sqrt{4t^2 + 1}}$. At $t=1$: $\boxed{4/\sqrt 5}$.
\item $y' = -1/x^2$, $y'' = 2/x^3$. $\kappa(x) = \dfrac{2/x^3}{(1 + 1/x^4)^{3/2}} = \dfrac{2 x^3}{(x^4 + 1)^{3/2}}$. At $x = 1$: $\kappa = \dfrac{2}{2^{3/2}} = \boxed{\tfrac{1}{\sqrt 2}}$.
\end{answers}
