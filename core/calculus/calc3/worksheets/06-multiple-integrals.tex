\wsheader{Calc 3}{Multiple Integrals}{A}
\begin{problems}
\item Evaluate $\displaystyle \int_0^1 \!\int_0^2 (x + 2y) \dy \dx$.
\item Sketch and evaluate $\displaystyle \iint_D x y \dA$ where $D$ is bounded by $y = x$ and $y = x^2$.
\item Convert to polar and evaluate $\displaystyle \iint_D e^{-(x^2+y^2)} \dA$ where $D$ is the disk $x^2 + y^2 \le 1$.
\item Evaluate $\displaystyle \iiint_E z \dV$ where $E = [0,1]^3$.
\item Use cylindrical coordinates to find the volume of the solid bounded above by $z = 4 - x^2 - y^2$ and below by $z = 0$.
\end{problems}

\wsheader{Calc 3}{Multiple Integrals}{B}
\begin{problems}
\item Evaluate $\displaystyle \int_0^1 \!\int_{x^2}^{x} (x^2 + y^2) \dy \dx$.
\item Reverse the order of integration: $\displaystyle \int_0^1 \!\int_y^1 \sin(x^2) \dx \dy$, then evaluate.
\item Convert to polar and evaluate $\displaystyle \iint_D x^2 \dA$ where $D$ is the upper half-disk $x^2 + y^2 \le 4, y \ge 0$.
\item Use spherical coordinates to compute the volume of the ball $x^2 + y^2 + z^2 \le 4$.
\item Find the volume of the solid in the first octant bounded by the plane $x + y + z = 1$.
\end{problems}

\anshdr{Calc 3}{Multiple Integrals}

\begin{answers}{A}
\item Inner: $\int_0^2 (x + 2y) \dy = 2x + 4$. Outer: $\int_0^1 (2x + 4) \dx = 1 + 4 = \boxed{5}$.
\item $D$: $0 \le x \le 1$, $x^2 \le y \le x$. $\iint xy \dA = \int_0^1 x \!\int_{x^2}^x y \dy \dx = \int_0^1 x \cdot \tfrac{x^2 - x^4}{2} \dx = \tfrac{1}{2}\!\int_0^1 (x^3 - x^5) \dx = \tfrac{1}{2}(1/4 - 1/6) = \boxed{1/24}$.
\item $\iint e^{-r^2} r \dr \dth = \int_0^{2\pi}\!\!\int_0^1 r e^{-r^2} \dr \dth = 2\pi \cdot [-\tfrac{1}{2} e^{-r^2}]_0^1 = \boxed{\pi(1 - e^{-1})}$.
\item $\int_0^1\!\int_0^1\!\int_0^1 z \dz \dy \dx = \int_0^1 \tfrac{1}{2} \dy \dx = \boxed{1/2}$.
\item $V = \int_0^{2\pi}\!\int_0^2\!\int_0^{4-r^2} r \dz \dr \dth = 2\pi \!\int_0^2 r(4 - r^2) \dr = 2\pi \cdot [2r^2 - r^4/4]_0^2 = 2\pi(8 - 4) = \boxed{8\pi}$.
\end{answers}

\begin{answers}{B}
\item Inner: $\int_{x^2}^x (x^2 + y^2) \dy = x^2(x - x^2) + \tfrac{x^3 - x^6}{3} = x^3 - x^4 + \tfrac{x^3 - x^6}{3} = \tfrac{4x^3 - 3x^4 - x^6}{3}$. Integrate: $\int_0^1 = \tfrac{1}{3}(1 - 3/5 - 1/7) = \tfrac{1}{3} \cdot \tfrac{35 - 21 - 5}{35} = \tfrac{9}{105} = \boxed{3/35}$.
\item Region: $0 \le y \le x \le 1$. Swap: $\int_0^1\!\int_0^x \sin(x^2) \dy \dx = \int_0^1 x \sin(x^2) \dx = \boxed{\tfrac{1}{2}(1 - \cos 1)}$.
\item $x^2 = r^2 \cos^2\theta$, $\dA = r \dr \dth$. $\int_0^\pi \cos^2\theta \dth \int_0^2 r^3 \dr = \tfrac{\pi}{2} \cdot 4 = \boxed{2\pi}$.
\item $V = \int_0^{2\pi}\!\int_0^\pi\!\int_0^2 \rho^2 \sin\varphi \drho \dphi \dth = 2\pi \cdot 2 \cdot \tfrac{8}{3} = \boxed{\tfrac{32\pi}{3}}$.
\item $V = \int_0^1\!\int_0^{1-x}\!\int_0^{1-x-y} 1 \dz \dy \dx = \int_0^1 \!\int_0^{1-x}(1-x-y) \dy \dx = \int_0^1 \tfrac{(1-x)^2}{2} \dx = \boxed{1/6}$.
\end{answers}
