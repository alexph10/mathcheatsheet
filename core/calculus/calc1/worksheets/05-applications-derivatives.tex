\begin{problems}
\item Find all critical points of $f(x) = x^3 - 3x^2 - 9x + 1$ and classify them.
\item Find the absolute extrema of $f(x) = x^3 - 12x$ on $[-3, 3]$.
\item A rectangle has perimeter $20$. Find the dimensions that maximize its area.
\item Verify MVT for $f(x) = x^2$ on $[1, 3]$ and find the $c$ guaranteed.
\item A ladder $10$ ft long leans against a wall. Its base slides out at $2$ ft/s. How fast is the top sliding down when the base is $6$ ft from the wall?
\end{problems}

\begin{problems}
\item Find the local extrema of $f(x) = \dfrac{x}{x^2 + 1}$.
\item Find the absolute extrema of $f(x) = x + 2 \sin x$ on $[0, 2\pi]$.
\item A box with a square base and open top has volume $32{,}000$ cm$^3$. Find the dimensions minimizing surface area.
\item Use linear approximation to estimate $\sqrt[3]{8.1}$.
\item Use one step of Newton's method on $f(x) = x^2 - 5$ starting from $x_0 = 2$ to estimate $\sqrt 5$.
\end{problems}

\anshdr{Calc 1}{Applications of the Derivative}

\begin{answers}{A}
\item $f' = 3x^2 - 6x - 9 = 3(x-3)(x+1) = 0$ at $x = -1, 3$. $f'' = 6x - 6$: $f''(-1) = -12 < 0$ (\textbf{local max}, value $6$); $f''(3) = 12 > 0$ (\textbf{local min}, value $-26$).
\item $f'(x) = 3x^2 - 12 = 0$ at $x = \pm 2$. Values: $f(-3) = -27 + 36 = 9$, $f(-2) = 16$, $f(2) = -16$, $f(3) = -9$. \textbf{Max $=16$ at $x=-2$; min $=-16$ at $x=2$.}
\item Let sides be $x, 10-x$. Area $A = x(10-x)$, $A' = 10 - 2x = 0 \Rightarrow x = 5$. So $\boxed{5 \times 5}$ (a square), area $25$.
\item $f$ is continuous on $[1,3]$, differentiable on $(1,3)$. Slope: $\tfrac{9-1}{2} = 4 = f'(c) = 2c \Rightarrow \boxed{c = 2}$.
\item $x^2 + y^2 = 100$, so $2x x' + 2y y' = 0$. At $x=6$: $y = 8$, $x' = 2$. Then $y' = -x x'/y = -12/8 = \boxed{-1.5}$ ft/s (top descends at $1.5$ ft/s).
\end{answers}

\begin{answers}{B}
\item $f' = \tfrac{(1)(x^2+1) - x(2x)}{(x^2+1)^2} = \tfrac{1 - x^2}{(x^2+1)^2}$. Zero at $x = \pm 1$. Sign change $+\to-$ at $x = 1$ (\textbf{max}, $f(1)=1/2$); $-\to+$ at $x = -1$ (\textbf{min}, $f(-1)=-1/2$).
\item $f'(x) = 1 + 2\cos x = 0 \Rightarrow \cos x = -1/2$, so $x = 2\pi/3, 4\pi/3$ in $[0,2\pi]$. Values: $f(0)=0$, $f(2\pi/3) = 2\pi/3 + \sqrt 3$, $f(4\pi/3) = 4\pi/3 - \sqrt 3$, $f(2\pi) = 2\pi$. \textbf{Max $= 2\pi$ at $x=2\pi$; min $=0$ at $x=0$.}
\item Let base $=x$, height $=h$. Volume $x^2 h = 32000 \Rightarrow h = 32000/x^2$. Surface area $S = x^2 + 4xh = x^2 + 128000/x$. $S' = 2x - 128000/x^2 = 0 \Rightarrow x^3 = 64000 \Rightarrow x = 40$, $h = 20$. \textbf{Dimensions $40 \times 40 \times 20$ cm.}
\item Near $a=8$: $f(x) = x^{1/3}$, $f'(x) = \tfrac{1}{3} x^{-2/3}$, $f'(8) = 1/12$. $\sqrt[3]{8.1} \approx 2 + \tfrac{0.1}{12} \approx \boxed{2.00833}$.
\item $x_1 = x_0 - \tfrac{f(x_0)}{f'(x_0)} = 2 - \tfrac{-1}{4} = \boxed{2.25}$ (true $\sqrt 5 \approx 2.2361$).
\end{answers}
