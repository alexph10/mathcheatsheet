\wsheader{Calc 1}{Techniques of Integration}{B}
\begin{problems}
\item Compute $\displaystyle \int_0^1 x e^{x^2} \dx$ ($u$-substitution).
\item Compute $\displaystyle \int \ln x \dx$ (integration by parts).
\item Compute $\displaystyle \int \dfrac{x^2}{\sqrt{1 - x^2}} \dx$ (trig substitution).
\item Determine whether $\displaystyle \int_1^\infty \dfrac{1}{x^2} \dx$ converges; if so, find its value.
\item Compute $\displaystyle \int \dfrac{2x + 1}{x^2 - x - 2} \dx$ (partial fractions).
\end{problems}

\anshdr{Calc 1}{Techniques of Integration}

\begin{answers}{A}
\item $u = x^2 + 1 \Rightarrow \du = 2x \dx$. Integral $= \int u^{1/2} \du = \tfrac{2}{3} u^{3/2} = \boxed{\tfrac{2}{3}(x^2+1)^{3/2} + C}$.
\item Parts: $u = x$, $dv = \cos x \dx \Rightarrow du = \dx$, $v = \sin x$. $\int x \cos x \dx = x \sin x - \int \sin x \dx = \boxed{x \sin x + \cos x + C}$.
\item $\sin^3 x = \sin x(1 - \cos^2 x)$. Let $u = \cos x$, $\du = -\sin x \dx$. Integral $= -\int (1 - u^2) u^2 \du = -\int (u^2 - u^4) \du = -\tfrac{u^3}{3} + \tfrac{u^5}{5} + C = \boxed{\tfrac{\cos^5 x}{5} - \tfrac{\cos^3 x}{3} + C}$.
\item Standard form $\boxed{\arcsin(x/2) + C}$.
\item $\tfrac{1}{(x-1)(x+2)} = \tfrac{1/3}{x-1} - \tfrac{1/3}{x+2}$. Integral $= \boxed{\tfrac{1}{3} \ln\!\left|\tfrac{x-1}{x+2}\right| + C}$.
\end{answers}

\begin{answers}{B}
\item $u = x^2$, $\du = 2x \dx$. $\int_0^1 x e^{x^2} \dx = \tfrac{1}{2} \int_0^1 e^u \du = \tfrac{1}{2}(e - 1) = \boxed{(e-1)/2}$.
\item Parts: $u = \ln x$, $dv = \dx \Rightarrow du = \tfrac{\dx}{x}$, $v = x$. $\int \ln x \dx = x \ln x - \int 1 \dx = \boxed{x \ln x - x + C}$.
\item $x = \sin\theta$, $\dx = \cos\theta \dth$, $\sqrt{1-x^2} = \cos\theta$. Integral $= \int \sin^2\theta \dth = \tfrac{1}{2}(\theta - \sin\theta \cos\theta) + C = \boxed{\tfrac{1}{2} \arcsin x - \tfrac{1}{2} x\sqrt{1-x^2} + C}$.
\item $\int_1^b \tfrac{\dx}{x^2} = [-1/x]_1^b = 1 - 1/b \to \boxed{1}$ as $b \to \infty$. Converges to $1$.
\item Factor: $x^2 - x - 2 = (x-2)(x+1)$. $\tfrac{2x+1}{(x-2)(x+1)} = \tfrac{A}{x-2} + \tfrac{B}{x+1}$. Solve: $A = 5/3$, $B = 1/3$. Integral $= \boxed{\tfrac{5}{3} \ln|x-2| + \tfrac{1}{3} \ln|x+1| + C}$.
\end{answers}
