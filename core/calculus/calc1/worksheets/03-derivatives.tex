\begin{problems}

\item Find the tangent line to $y = \sqrt{x}$ at $x = 4$.

\wsheader{Calc 1}{The Derivative}{B}
\begin{problems}
\item Use the limit definition to compute $f'(x)$ for $f(x) = 1/x$.
\item Find the equations of both the tangent and the normal line to $y = x^2 + 1$ at $x = 1$.
\item Show $f(x) = x^{1/3}$ has a vertical tangent at $x = 0$.
\item For $f(x) = \begin{cases} x^2, & x \le 1 \\ 2x - 1, & x > 1\end{cases}$, is $f$ differentiable at $x = 1$?
\item A balloon rises so its height (m) at time $t$ (s) is $h(t) = 5t + t^2$. Find its velocity at $t = 3$ and the average velocity on $[1, 3]$.
\end{problems}

\anshdr{Calc 1}{The Derivative}

\begin{answers}{A}
\item $\tfrac{3(x+h)^2 - (x+h) - (3x^2 - x)}{h} = \tfrac{6xh + 3h^2 - h}{h} = 6x + 3h - 1 \to \boxed{6x - 1}$.
\item $y'(x) = \tfrac{1}{2\sqrt x}$, so slope at $x=4$ is $1/4$; point $(4,2)$. Tangent: $\boxed{y = 2 + \tfrac{1}{4}(x - 4)}$.
\item Left slope $\lim_{h\to 0^-} \tfrac{|2+h-2|}{h} = \tfrac{-h}{h} = -1$; right slope $=+1$. Slopes differ $\Rightarrow$ $f'(2)$ \textbf{does not exist}.
\item $\lim_{h\to 0} \tfrac{(1+h)^3 - 2(1+h) - (-1)}{h} = \lim \tfrac{3h + 3h^2 + h^3 - 2h}{h} = \lim(1 + 3h + h^2) = \boxed{1}$.
\item $v(t) = 3t^2 - 12t + 9 = 3(t-1)(t-3)$, $a(t) = 6t - 12$. Particle at rest when $v=0$: $\boxed{t = 1, 3}$.
\end{answers}

\begin{answers}{B}
\item $\tfrac{1/(x+h) - 1/x}{h} = \tfrac{-h}{h\, x(x+h)} = \tfrac{-1}{x(x+h)} \to \boxed{-1/x^2}$.
\item $y'(1) = 2$. Tangent: $y = 2 + 2(x-1)$. Normal slope $-1/2$: $y = 2 - \tfrac{1}{2}(x-1)$.
\item $f'(x) = \tfrac{1}{3} x^{-2/3} \to +\infty$ as $x \to 0$. So tangent at $(0,0)$ is vertical.
\item Continuity: both pieces give $1$ at $x=1$. Left derivative $=2x|_{x=1} = 2$; right derivative $=2$. Equal $\Rightarrow$ \textbf{differentiable}; $f'(1) = 2$.
\item $h'(t) = 5 + 2t$, so $h'(3) = \boxed{11}$ m/s. Avg velocity $= \tfrac{h(3)-h(1)}{3-1} = \tfrac{24 - 6}{2} = \boxed{9}$ m/s.
\end{answers}
