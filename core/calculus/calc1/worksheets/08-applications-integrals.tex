\wsheader{Calc 1}{Applications of the Integral}{A}
\begin{problems}
\item Find the area between $y = x^2$ and $y = 2x$ on $[0, 2]$.
\item The region under $y = \sqrt{x}$, $x \in [0, 4]$, is revolved about the $x$-axis. Find the volume (disks).
\item Find the arc length of $y = \tfrac{2}{3} x^{3/2}$ on $[0, 3]$.
\item Find the average value of $f(x) = \sin x$ on $[0, \pi]$.
\item Solve the separable ODE $\dfrac{dy}{dx} = xy$, $y(0) = 2$.
\end{problems}

\wsheader{Calc 1}{Applications of the Integral}{B}
\begin{problems}
\item Find the area enclosed between $y = x^2$ and $y = x + 2$.
\item The region under $y = x$, $x \in [0, 1]$, is revolved about the $y$-axis. Find the volume (shells).
\item Find the arc length of $y = \ln(\cos x)$ on $[0, \pi/4]$.
\item A force $F(x) = 10 - x$ acts on an object moving from $x = 0$ to $x = 5$. Find the work done.
\item Solve the IVP $\dfrac{dy}{dx} = \dfrac{y}{x}$, $y(1) = 3$ (with $x > 0$).
\end{problems}

\anshdr{Calc 1}{Applications of the Integral}

\begin{answers}{A}
\item Intersection: $x^2 = 2x \Rightarrow x = 0, 2$. On $[0,2]$, $2x \ge x^2$. $A = \int_0^2 (2x - x^2) \dx = [x^2 - x^3/3]_0^2 = 4 - 8/3 = \boxed{4/3}$.
\item $V = \pi \int_0^4 (\sqrt x)^2 \dx = \pi \int_0^4 x \dx = \pi \cdot 8 = \boxed{8\pi}$.
\item $y' = \sqrt x$. $L = \int_0^3 \sqrt{1 + x} \dx = \bigl[\tfrac{2}{3}(1+x)^{3/2}\bigr]_0^3 = \tfrac{2}{3}(8 - 1) = \boxed{14/3}$.
\item $\bar f = \tfrac{1}{\pi} \int_0^\pi \sin x \dx = \tfrac{1}{\pi} \cdot 2 = \boxed{2/\pi}$.
\item Separate: $\tfrac{dy}{y} = x \dx \Rightarrow \ln|y| = \tfrac{x^2}{2} + C \Rightarrow y = A e^{x^2/2}$. $y(0) = 2 \Rightarrow A = 2$. $\boxed{y = 2 e^{x^2/2}}$.
\end{answers}

\begin{answers}{B}
\item Intersect: $x^2 = x + 2 \Rightarrow x = -1, 2$. $A = \int_{-1}^{2} (x + 2 - x^2) \dx = [x^2/2 + 2x - x^3/3]_{-1}^2 = \tfrac{10}{3} - (-\tfrac{7}{6}) = \boxed{9/2}$.
\item Shells about $y$-axis: $V = 2\pi \int_0^1 x \cdot x \dx = 2\pi \cdot \tfrac{1}{3} = \boxed{2\pi/3}$.
\item $y' = -\tan x$, $1 + (y')^2 = 1 + \tan^2 x = \sec^2 x$. $L = \int_0^{\pi/4} \sec x \dx = [\ln|\sec x + \tan x|]_0^{\pi/4} = \boxed{\ln(\sqrt 2 + 1)}$.
\item $W = \int_0^5 (10 - x) \dx = [10x - x^2/2]_0^5 = 50 - 12.5 = \boxed{37.5}$.
\item Separate: $\tfrac{dy}{y} = \tfrac{dx}{x} \Rightarrow \ln y = \ln x + C \Rightarrow y = A x$. $y(1) = 3 \Rightarrow A = 3$. $\boxed{y = 3x}$.
\end{answers}
