\wsheader{Calc 1}{The Integral}{A}
\begin{problems}
\item Compute $\displaystyle \int (3x^2 - 4x + 1) \dx$.
\item Compute $\displaystyle \int_0^1 (e^x + 1) \dx$.
\item Compute $\displaystyle \int \!\Bigl(\tfrac{1}{x} + \sin x\Bigr) \dx$.
\item Find $F'(x)$ where $F(x) = \displaystyle \int_0^{x^2} \cos(t) \dt$.
\item Find the average value of $f(x) = x^2$ on $[0, 3]$.
\end{problems}

\wsheader{Calc 1}{The Integral}{B}
\begin{problems}
\item Compute $\displaystyle \int_0^4 \sqrt{x} \dx$.
\item Compute $\displaystyle \int_{-1}^{1} (x^3 + x) \dx$.
\item Compute $\displaystyle \int \sec^2 x \dx$.
\item Find $F'(x)$ where $F(x) = \displaystyle \int_x^{x^2} e^{-t^2} \dt$.
\item Approximate $\displaystyle \int_0^4 (x^2 + 1) \dx$ using the right Riemann sum with $n=4$.
\end{problems}

\anshdr{Calc 1}{The Integral}

\begin{answers}{A}
\item $\boxed{x^3 - 2x^2 + x + C}$.
\item $[e^x + x]_0^1 = (e + 1) - 1 = \boxed{e}$.
\item $\boxed{\ln|x| - \cos x + C}$.
\item Leibniz: $F'(x) = \cos(x^2) \cdot 2x = \boxed{2x \cos(x^2)}$.
\item $\bar f = \tfrac{1}{3}\!\int_0^3 x^2 \dx = \tfrac{1}{3} \cdot 9 = \boxed{3}$.
\end{answers}

\begin{answers}{B}
\item $\bigl[\tfrac{2}{3} x^{3/2}\bigr]_0^4 = \tfrac{2}{3} \cdot 8 = \boxed{16/3}$.
\item Odd integrand on symmetric interval: $\boxed{0}$ (or compute $[\tfrac{x^4}{4} + \tfrac{x^2}{2}]_{-1}^{1} = 0$).
\item $\boxed{\tan x + C}$.
\item Leibniz: $F'(x) = e^{-x^4} \cdot 2x - e^{-x^2} \cdot 1 = \boxed{2x e^{-x^4} - e^{-x^2}}$.
\item $\Delta x = 1$, sample at $x_i = 1,2,3,4$. Sum $= 1(2 + 5 + 10 + 17) = \boxed{34}$ (exact value $= 4^3/3 + 4 = 76/3 \approx 25.33$).
\end{answers}
